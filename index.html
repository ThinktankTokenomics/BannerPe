// BannerSpace Escrow Contract
contract BannerEscrow {
    address public advertiser;
    address public influencer;
    uint256 public amount;
    uint256 public campaignEnd;
    
    // Funds automatically locked on creation
    constructor(address _influencer) payable {
        advertiser = msg.sender;
        influencer = _influencer;
        amount = msg.value;
    }
    
    // Only release when oracle verifies completion
    function releasePayment(bytes memory oracleProof) external {
        require(block.timestamp >= campaignEnd, "Campaign ongoing");
        require(verifyOracle(oracleProof), "Invalid verification");
        
        // 90% to influencer, 10% platform fee
        payable(influencer).transfer(amount * 90 / 100);
        payable(platform).transfer(amount * 10 / 100);
    }
    
    // Automatic refund if influencer rejects/ignores
    function refundIfNoResponse() external {
        require(block.timestamp > responseDeadline, "Response period active");
        payable(advertiser).transfer(amount);
    }
}
